40void UHS2_GPIO::digitalWrite(uint8_t pin, uint8_t val) {
41 if(pin > 7)
42 return;
43 uint8_t nValue = m_pUsb->gpioRdOutput();
44 uint8_t nMask = 1 << pin;
45 nValue &= (~nMask);
46 if(val)
47 nValue |= (nMask);
48 m_pUsb->gpioWr(nValue);
49}

55int UHS2_GPIO::digitalRead(uint8_t pin) {
