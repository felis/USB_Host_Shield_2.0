65  bool Skip(uint8_t **pp, uint16_t *pcntdn, uint16_t bytes_to_skip) {
66  switch(nStage) {
67  case 0:
68  countDown = bytes_to_skip;
69  nStage++;
70  case 1:
71  for(; countDown && (*pcntdn); countDown--, (*pp)++, (*pcntdn)--);
72 
73  if(!countDown)
74  nStage = 0;
75  };
76  return (!countDown);
77  };
