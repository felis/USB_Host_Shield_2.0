77  bool Skip(uint8_t **pp, uint16_t *pcntdn, uint16_t bytes_to_skip) {
78  switch(nStage) {
79  case 0:
80  countDown = bytes_to_skip;
81  nStage++;
82  // fall through
83  case 1:
84  for(; countDown && (*pcntdn); countDown--, (*pp)++, (*pcntdn)--);
85 
86  if(!countDown)
87  nStage = 0;
88  };
89  return (!countDown);
90  };
91 };
