417 CommandBlockWrapperBase(tag, xflen, flgs),
418 bmCBWLUN(lu), bmReserved1(0), bmCBWCBLength(cmdlen), bmReserved2(0) {
419 for(int i = 0; i < 16; i++) CBWCB[i] = 0;
420 // Type punning can cause optimization problems and bugs.

424 x->LUN = cmd;
425 }
