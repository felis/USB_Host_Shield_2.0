81 struct ps5TouchpadXY {
82  struct {
83  uint8_t counter : 7; // Increments every time a finger is touching the touchpad
84  uint8_t touching : 1; // The top bit is cleared if the finger is touching the touchpad
85  uint16_t x : 12;
