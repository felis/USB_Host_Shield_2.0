1098  else {

1100 
1101  itemPrefix = (**pp);

1120  theBuffer.valueSize = itemSize;
1121  valParser.Initialize(&theBuffer);
1122  itemParseState = 2;
1123  // fall through
1124  case 2:
1125  if(!valParser.Parse(pp, pcntdn))
1126  return enErrorIncomplete;
1127  itemParseState = 3;
1128  // fall through

1130  {
