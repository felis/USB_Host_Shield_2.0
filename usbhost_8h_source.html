183/* write single byte into MAX3421 register */
184template< typename SPI_SS, typename INTR >
185void MAX3421e< SPI_SS, INTR >::regWr(uint8_t reg, uint8_t data) {
186 XMEM_ACQUIRE_SPI();
187#if defined(SPI_HAS_TRANSACTION)
188 USB_SPI.beginTransaction(SPISettings(26000000, MSBFIRST, SPI_MODE0)); // The MAX3421E can handle up to 26MHz, use MSB First and SPI mode 0
189#endif
190 SPI_SS::Clear();
191
192#if USING_SPI4TEENSY3
193 uint8_t c[2];
194 c[0] = reg | 0x02;
195 c[1] = data;
196 spi4teensy3::send(c, 2);
197#elif defined(SPI_HAS_TRANSACTION) && !defined(ESP8266) && !defined(ESP32)
198 uint8_t c[2];
199 c[0] = reg | 0x02;
200 c[1] = data;
201 USB_SPI.transfer(c, 2);
202#elif defined(STM32F4)
203 uint8_t c[2];
204 c[0] = reg | 0x02;
205 c[1] = data;
206 HAL_SPI_Transmit(&SPI_Handle, c, 2, HAL_MAX_DELAY);
207#elif !defined(SPDR) // ESP8266, ESP32
208 USB_SPI.transfer(reg | 0x02);
209 USB_SPI.transfer(data);
210#else
211 SPDR = (reg | 0x02);
212 while(!(SPSR & (1 << SPIF)));
213 SPDR = data;
214 while(!(SPSR & (1 << SPIF)));
215#endif
216
217 SPI_SS::Set();
218#if defined(SPI_HAS_TRANSACTION)
219 USB_SPI.endTransaction();
220#endif
221 XMEM_RELEASE_SPI();
222 return;
223};

226/* returns a pointer to memory position after last written */
227template< typename SPI_SS, typename INTR >
228uint8_t* MAX3421e< SPI_SS, INTR >::bytesWr(uint8_t reg, uint8_t nbytes, uint8_t* data_p) {
229 XMEM_ACQUIRE_SPI();
230#if defined(SPI_HAS_TRANSACTION)
231 USB_SPI.beginTransaction(SPISettings(26000000, MSBFIRST, SPI_MODE0)); // The MAX3421E can handle up to 26MHz, use MSB First and SPI mode 0
232#endif
233 SPI_SS::Clear();
234
235#if USING_SPI4TEENSY3
236 spi4teensy3::send(reg | 0x02);
237 spi4teensy3::send(data_p, nbytes);
238 data_p += nbytes;
239#elif defined(STM32F4)
240 uint8_t data = reg | 0x02;
241 HAL_SPI_Transmit(&SPI_Handle, &data, 1, HAL_MAX_DELAY);
242 HAL_SPI_Transmit(&SPI_Handle, data_p, nbytes, HAL_MAX_DELAY);
243 data_p += nbytes;
244#elif !defined(__AVR__) || !defined(SPDR)
245#if defined(ESP8266) || defined(ESP32)
246 yield();
247#endif
248 USB_SPI.transfer(reg | 0x02);
249 while(nbytes) {
250 USB_SPI.transfer(*data_p);
251 nbytes--;
252 data_p++; // advance data pointer
253 }
254#else
255 SPDR = (reg | 0x02); //set WR bit and send register number
256 while(nbytes) {
257 while(!(SPSR & (1 << SPIF))); //check if previous byte was sent
258 SPDR = (*data_p); // send next data byte
259 nbytes--;
260 data_p++; // advance data pointer
261 }
262 while(!(SPSR & (1 << SPIF)));
263#endif
264
265 SPI_SS::Set();
266#if defined(SPI_HAS_TRANSACTION)
267 USB_SPI.endTransaction();
268#endif
269 XMEM_RELEASE_SPI();
270 return ( data_p);
271}

275/* GPOUT bits are in the low nibble. 0-3 in IOPINS1, 4-7 in IOPINS2 */
276template< typename SPI_SS, typename INTR >
277void MAX3421e< SPI_SS, INTR >::gpioWr(uint8_t data) {
278 regWr(rIOPINS1, data);
279 data >>= 4;
280 regWr(rIOPINS2, data);
281 return;
282}

391template< typename SPI_SS, typename INTR >
392uint8_t MAX3421e< SPI_SS, INTR >::gpioRd() {
393 uint8_t gpin = 0;
394 gpin = regRd(rIOPINS2); //pins 4-7
395 gpin &= 0xf0; //clean lower nibble
396 gpin |= (regRd(rIOPINS1) >> 4); //shift low bits and OR with upper from previous operation.
397 return ( gpin);
398}

403/* GPOUT pins are in low nibbles of IOPINS1, IOPINS2 */
404template< typename SPI_SS, typename INTR >
405uint8_t MAX3421e< SPI_SS, INTR >::gpioRdOutput() {
406 uint8_t gpout = 0;
407 gpout = regRd(rIOPINS1); //pins 0-3
408 gpout &= 0x0f; //clean upper nibble
409 gpout |= (regRd(rIOPINS2) << 4); //shift high bits and OR with lower from previous operation.
410 return ( gpout);

428/* initialize MAX3421E. Set Host mode, pullups, and stuff. Returns 0 if success, -1 if not */
429template< typename SPI_SS, typename INTR >
430int8_t MAX3421e< SPI_SS, INTR >::Init() {
431 XMEM_ACQUIRE_SPI();
432 // Moved here.
433 // you really should not init hardware in the constructor when it involves locks.
434 // Also avoids the vbus flicker issue confusing some devices.
435 /* pin and peripheral setup */
436 SPI_SS::SetDirWrite();
437 SPI_SS::Set();
438 spi::init();
439 INTR::SetDirRead();
440 XMEM_RELEASE_SPI();
441 /* MAX3421E - full-duplex SPI, level interrupt */
442 // GPX pin on. Moved here, otherwise we flicker the vbus.
443 regWr(rPINCTL, (bmFDUPSPI | bmINTLEVEL));
444
445 if(reset() == 0) { //OSCOKIRQ hasn't asserted in time
446 return ( -1);
447 }
448
449 regWr(rMODE, bmDPPULLDN | bmDMPULLDN | bmHOST); // set pull-downs, Host
450
451 regWr(rHIEN, bmCONDETIE | bmFRAMEIE); //connection detection
452
453 /* check if device is connected */
454 regWr(rHCTL, bmSAMPLEBUS); // sample USB bus
455 while(!(regRd(rHCTL) & bmSAMPLEBUS)); //wait for sample operation to finish
456
457 busprobe(); //check if anything is connected
458
459 regWr(rHIRQ, bmCONDETIRQ); //clear connection detect interrupt
460 regWr(rCPUCTL, 0x01); //enable interrupt pin
461
462 return ( 0);
463}

465/* initialize MAX3421E. Set Host mode, pullups, and stuff. Returns 0 if success, -1 if not */
466template< typename SPI_SS, typename INTR >
467int8_t MAX3421e< SPI_SS, INTR >::Init(int mseconds) {
468 XMEM_ACQUIRE_SPI();
469 // Moved here.
470 // you really should not init hardware in the constructor when it involves locks.
471 // Also avoids the vbus flicker issue confusing some devices.
472 /* pin and peripheral setup */
473 SPI_SS::SetDirWrite();
474 SPI_SS::Set();
475 spi::init();
476 INTR::SetDirRead();
477 XMEM_RELEASE_SPI();
478 /* MAX3421E - full-duplex SPI, level interrupt, vbus off */
479 regWr(rPINCTL, (bmFDUPSPI | bmINTLEVEL | GPX_VBDET));
480
481 if(reset() == 0) { //OSCOKIRQ hasn't asserted in time
482 return ( -1);
483 }
484
485 // Delay a minimum of 1 second to ensure any capacitors are drained.
486 // 1 second is required to make sure we do not smoke a Microdrive!
487 if(mseconds < 1000) mseconds = 1000;
488 delay(mseconds);
489
490 regWr(rMODE, bmDPPULLDN | bmDMPULLDN | bmHOST); // set pull-downs, Host
491
492 regWr(rHIEN, bmCONDETIE | bmFRAMEIE); //connection detection
493
494 /* check if device is connected */
495 regWr(rHCTL, bmSAMPLEBUS); // sample USB bus
496 while(!(regRd(rHCTL) & bmSAMPLEBUS)); //wait for sample operation to finish
497
498 busprobe(); //check if anything is connected
499
500 regWr(rHIRQ, bmCONDETIRQ); //clear connection detect interrupt
501 regWr(rCPUCTL, 0x01); //enable interrupt pin
502
503 // GPX pin on. This is done here so that busprobe will fail if we have a switch connected.
504 regWr(rPINCTL, (bmFDUPSPI | bmINTLEVEL));
505
506 return ( 0);
507}

509/* probe bus to determine device presence and speed and switch host to this speed */
510template< typename SPI_SS, typename INTR >
511void MAX3421e< SPI_SS, INTR >::busprobe() {
512 uint8_t bus_sample;
513 bus_sample = regRd(rHRSL); //Get J,K status
514 bus_sample &= (bmJSTATUS | bmKSTATUS); //zero the rest of the byte
515 switch(bus_sample) { //start full-speed or low-speed host
516 case( bmJSTATUS):
517 if((regRd(rMODE) & bmLOWSPEED) == 0) {
518 regWr(rMODE, MODE_FS_HOST); //start full-speed host
519 vbusState = FSHOST;
520 } else {
521 regWr(rMODE, MODE_LS_HOST); //start low-speed host
522 vbusState = LSHOST;
523 }
524 break;
525 case( bmKSTATUS):
526 if((regRd(rMODE) & bmLOWSPEED) == 0) {
527 regWr(rMODE, MODE_LS_HOST); //start low-speed host
528 vbusState = LSHOST;
529 } else {
530 regWr(rMODE, MODE_FS_HOST); //start full-speed host
531 vbusState = FSHOST;
532 }
533 break;
534 case( bmSE1): //illegal state
535 vbusState = SE1;
536 break;
537 case( bmSE0): //disconnected state
538 regWr(rMODE, bmDPPULLDN | bmDMPULLDN | bmHOST | bmSEPIRQ);
539 vbusState = SE0;
540 break;
541 }//end switch( bus_sample )
542}
