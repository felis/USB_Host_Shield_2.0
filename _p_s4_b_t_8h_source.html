83  buf[0] = 0x52; // HID BT Set_report (0x50) | Report Type (Output 0x02)
84  buf[1] = 0x11; // Report ID

97 
98  output->reportChanged = false;
99

101 
102  HID_Command(buf, sizeof(buf));
